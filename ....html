  1.   Построение модели в  Google BigQuery

        

        Для построения модели в Google BigQuery данные для моделирования и для предсказания были загружены из  Knime.

        Создаем модель градиентного бустинга GBM - BOOSTED_TREE_REGRESSOR с помощью библиотеки XGBoost

CREATE OR REPLACE MODEL`my-project-1-jule.mySchema.gbm`

OPTIONS(MODEL_TYPE='BOOSTED_TREE_REGRESSOR',

        BOOSTER_TYPE = 'DART',

        NUM_PARALLEL_TREE = 1,

        MAX_ITERATIONS = 300,

        TREE_METHOD = 'AUTO',

        EARLY_STOP = FALSE,

        SUBSAMPLE = 0.85,

        ENABLE_GLOBAL_EXPLAIN = TRUE,

        INPUT_LABEL_COLS =['Sales']

) AS

SELECT

  Sales AS Sales,  ProductID, Quantity, SupplierID, CategoryID, EmployeeID, CustomerID, RFM

FROM

 `my-project-1-jule.mySchema.DF_train`

Оценка модели

Select

*

from

ML.EVALUATE(MODEL `my-project-1-jule.mySchema.gbm`)
